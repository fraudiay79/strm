    var timeoutOverlay = document.getElementById('timeout-overlay');

    var timeoutOverlayVisible = false;

    var timeoutElapsedSeconds = 0;

    var timeoutMaxTime = 3600;

    function displayTimeoutOverlay(){
        timeoutOverlay.style.display = 'block';
        lightcastOverlays.style.visibility = 'visible';
                $('#lightcast-overlays').animate({'width': '100%'},300);
                showOverlay(true);
        timeoutOverlayVisible = true;
    }

    function closeTimeoutOverlay(){
        $('#lightcast-overlays').animate({'width': 0},300,function(){
            timeoutOverlay.style.display = 'none';
            lightcastOverlays.style.visibility = 'hidden';
            showOverlay(false);
        });
        timeoutOverlayVisible = false;
    }

    lightcastPlayer.on('displayClick', function(){
        //('Display Clicked - Resetting Timeout Counter');
        timeoutElapsedSeconds = 0;
    });

    lightcastPlayer.on('ready', function(){
        $( ".jw-controlbar" ).click(function() {
            //('Controlbar Clicked - Resetting Timeout Counter');
            timeoutElapsedSeconds = 0;
            if(timeoutOverlayVisible){
                closeTimeoutOverlay();
            }
        });

        setInterval(function(){
            if(lightcastPlayer.getState() != 'playing'){
                return;
            }
            timeoutElapsedSeconds++;
            //('Timeout Counter: ' + timeoutElapsedSeconds + ' of ' + timeoutMaxTime);
            if(timeoutElapsedSeconds >= timeoutMaxTime){
                lightcastPlayer.pause(true);
                                displayTimeoutOverlay();
                timeoutElapsedSeconds = 0;
            }
        }, 1000);

    });
